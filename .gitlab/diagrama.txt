    class Gerenciador {
        +estoque: Estoque
        clientes: Cliente[]
        reservas: Reserva[]
        locacoes: Reserva[]
        pagamentos: Pagamento[]

        cadastra_cliente(Cliente): void
        atualiza_cliente(Cliente, cpf, nome): void
        remove_cliente(Cliente): void

        cadastra_reserva(Reserva)
        cancela_reserva(Reserva)
        inicia_locacao(Reserva)
        finaliza_locacao(Locacao)

        gera_tabela_precos(): string
        gera_relatorio_mensal(mes): string
        gera_veiculos_mais_queridos(): string
    }

    class Estoque {
        veiculos: Veiculo[]
        cadastra_veiculo(Veiculo): void
        remove_veiculo(Veiculo): void
        veiculos_disponivels(data_inicio, data_fim): Veiculo[]
    }

    class Cliente {
        +cpf: string
        +nome: string
        reservas: Reserva[]
        atualiza_cadastro()
    }

    class Veiculo {
        +placa: string
        +fabricante: string
        +modelo: string
        +ano: integer
        +diaria_padrao: float
        diaria_desconto: float
        reservas: Reservas[]
        locacao: Locacao[]
    }

    class Pagamento {
        +reserva: Locacao
        +dateFormat:data
        +float: valor
    }

    class Reserva {
        +cliente: Cliente
        +veiculo: Veiculo
        +data_inicio: dateFormat
        +data_fim: dateFormat
        +preco: float
    }

    class Locacao {
        +cliente: Cliente
        +veiculo: Veiculo
        +data_inicio: dateFormat
        +data_fim: dateFormat
        +preco: float
        realiza_pagamento(): Pagamento
    }
